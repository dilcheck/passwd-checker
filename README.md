## except rule
1. words
> 'love', 'happy', 'qwer', 'asdf', 'zxcv', 'test', 'gpin', 'ipin'
2. email form
> '####@####.###'
3. special character
> '<', '>', '(', ')', '#', '\'', '/', '|'
